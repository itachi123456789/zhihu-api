zhihu API (v1.0.0)

user/org.followees

user/org.followees([offset])

Get a page of (at most 20) followees with given offset. It will get the first page if offset is not specified.

For example:

api.user('zhihuadmin')
    .followees()
    .then(console.log)
    .catch(console.trace)

the result is:

It's also easy to get all followees:

var user = api.user('zhihuadmin')
var offset = 0

next()

function next() {
    user.followees(offset)
        .then(data => {
            if (data.length) {
                offset += data.length
                handleFollowees(data)
                next()
            } else {
                console.log('Done!')
            }
        })
        .catch(console.trace)
}

    // do something
