zhihu API (v1.0.0)

topic.topAnswers

topic.topAnswers([page])

Get top answers (or 精华) under this topic. By default, page is 1.

For example:

api.topic(19550429)
    .topAnswers()
    .then(console.log)
    .catch(console.trace)

the result is:

It's also easy to get all top answers under this topic:

var topic = api.topic(19550429)
var page = 1

next()

function next() {
    topic.topAnswers(page)
        .then(data => {
            if (data.length) {
                page++
                handleAnswers(data)
                next()
            } else {
                console.log('Done!')
            }
        })
        .catch(console.trace)
}

    // do something
